# 安裝 virtualenv 工具
pip install virtualenv

# 使用 virtualenv 創建一個名為 venv 的虛擬環境
virtualenv venv

# 在 PowerShell 中刪除虛擬環境 (請注意：Windows 環境專用)
Remove-Item -Recurse -Force .\venv

# 將已安裝的套件及其版本信息凍結到 requirements.txt 文件中
pip freeze > requirements.txt

# 從 requirements.txt 安裝依賴套件到虛擬環境中
pip install -r requirements.txt
